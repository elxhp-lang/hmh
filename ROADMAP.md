async step8End(workflowId: string): Promise<ProcessingResult> {
  const workflow = this.getWorkflow(workflowId);

  // 1. 蒸馏用户偏好
  const preferences = {
    preferredStyle: workflow.selectedScript?.style,
    preferredPlatform: workflow.selectedScript?.platform,
    preferredDuration: workflow.selectedScript?.duration,
    lastUsedAt: new Date().toISOString(),
  };

  // 2. 保存偏好数据
  await this.saveUserPreferences(workflow.userId, preferences);

  // 3. 结束工作流
  workflow.status = 'completed';

  return {
    type: 'text',
    content: '工作流完成，已记录您的偏好',
  };
}
